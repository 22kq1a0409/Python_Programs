Problem 1 : program to print sum prime numbers b/w m and n
m=int(input())
n=int(input())
sum=0
for i in range(m,n):
    count=0
    for j in range(1,i+1):
        if(i%j==0):
            count+=1
    if(count==2):
        print(i)
        sum+=i
print(sum)
