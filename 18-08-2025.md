Problem 1 :Program to check if sum of individual digits of a number is prime or not

n=int(input())
sum=0
while(n>0):
    rem=n%10
    sum=sum+rem
    n=n//10
print(sum)

Problem 2 : Program to find given number is strong number or not

def fact (n):
    f=1
    for i in range(1,n+1):
        f=f*i
    return f
n=int(input())
sum=0
for i in str(n):
    sum+=fact(int(i))
if sum==n:
    print("strong")
else:
    print("not")
